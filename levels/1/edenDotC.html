<#include God.c>
    <#include earth.c>

    int main() {
      garden* eden = earth.eden;
      person* adam = God.breathe_life(dust);
      tree* theTree = eden.treeOfKnowledge;
      assert(adam.eat(theTree.fruit) == false);
      while (adam.working()) {
        creature* animal = God.choose_from(earth.animals);
        adam.give_name(animal);
        if (adam.lonely) {
          break;
        }
      }
      while (adam.sleeping()) {
        person* eve = God.breathe_life(adam.rib);
      }
      adam.join(eve);
      assert(adam.naked && adam.unashamed);
    }
